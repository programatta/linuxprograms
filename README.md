## git and git-lfs
~~~~
sudo apt-get update
sudo apt-get install git
sudo curl -s https://packagecloud.io/install/repositories/github/git-lfs/script.deb.sh | sudo bash
sudo apt-get install git-lfs
git lfs install
~~~~
